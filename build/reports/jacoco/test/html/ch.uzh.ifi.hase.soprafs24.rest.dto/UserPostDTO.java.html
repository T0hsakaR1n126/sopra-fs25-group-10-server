package ch.uzh.ifi.hase.soprafs24.rest.dto;

public class UserPostDTO {

  private String token;

  private Long userId;

    return name;
  }

  public void setName(String name) {
    this.name = name;
  }

    return username;

  }

    return password;

    this.password = password;
  }

  public Long getUserId() {
    return userId;
  }

  public void setUserId(Long userId) {
    this.userId = userId;
  }
}
